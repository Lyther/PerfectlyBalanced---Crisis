# Initialize event
event = {
    id = pbcrisis.0
    hide_window = yes
    is_triggered_only = yes
    immediate = {
        if = {
            limit = {
                NOT = {
                    has_global_flag = pbcrisis_initialized
                }

            set_global_flag = pbcrisis_initialized

            set_global_flag = pbcrisis_general_difficulty_1

    }
}

# Detection of final crisis
